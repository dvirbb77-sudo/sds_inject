INSTALLATION FLOW DIAGRAM
================================================================================

User runs installer
        ↓
[Validate system]
  ├─ OS check (Ubuntu 22.04)
  ├─ Root access check
  ├─ CPU/Memory/Disk check
  ├─ Network connectivity
  └─ Required commands check
        ↓
[Detect node type]
  ├─ Is Kubernetes installed?
  ├─ Is it master or worker?
  └─ What's the current state?
        ↓
[Choose installation mode]
  │
  ├─→ [MASTER MODE]
  │     ├─ Load kernel modules
  │     ├─ Configure sysctl
  │     ├─ Install containerd
  │     ├─ Install Kubernetes packages
  │     ├─ Run kubeadm init
  │     └─ Setup kubectl access
  │
  ├─→ [WORKER MODE]
  │     ├─ Load kernel modules
  │     ├─ Configure sysctl
  │     ├─ Install containerd
  │     ├─ Install Kubernetes packages
  │     ├─ Run kubeadm join
  │     └─ Register with cluster
  │
  └─→ [AUTO MODE]
        If uninitialized: Install as MASTER
        If master: Warn about dangers
        If worker: Offer upgrade options
        ↓
[Installation complete]
  ├─ Logs saved to: logs/install-YYYYMMDD-HHMMSS.log
  ├─ Node ready for workload deployment
  └─ Verify: kubectl get nodes

BUILD COMMANDS:

  make lint              # Run shellcheck
  make test              # Run unit tests
  make build             # Create installer artifact
  make test-vm           # Boot Vagrant VM and test
  make clean             # Remove build artifacts

INSTALLER MODES:
  # Auto-detect and install
  k8s-installer.run

  # Install as master
  k8s-installer.run --master

  # Install as worker
  k8s-installer.run --worker --master-ip 192.168.1.100 --join-token abc123

  # Validate only (dry-run)
  k8s-installer.run --validate-only

  # Enable debug logging
  DEBUG=1 k8s-installer.run --master

DEPLOYMENT:
  # Copy and run directly
  scp dist/k8s-installer.run root@target:/tmp/
  ssh root@target /tmp/k8s-installer.run --master

  # Use deploy script
  cd/deploy.sh --host 192.168.1.10 --mode master

  # Reconciliation
  cd/reconcile.sh

VERIFICATION:
  # Check logs
  tail -f logs/install-*.log

  # Check Kubernetes
  kubectl get nodes
  kubectl cluster-info
  kubectl get pods -A

  # Check services
  systemctl status kubelet
  systemctl status containerd
